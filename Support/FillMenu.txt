For i = 0 To 26
   mEditor(i).Caption = PublicTags(i)
   
   If i = 0 Or i = Tag_Register Or i = Tag_Variable Or i = Tag_Local_Variable Or i = Tags_End Then
     mEditor(i).Visible = False
   Else

           For j = N_Item To mItm.ubound
